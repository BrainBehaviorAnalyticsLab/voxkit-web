voxkit.analyzers

VoxKit Analyzers Module.

Analyzers observe datasets and link metadata in flexible, abstract ways. They extract structured information at registration time, producing CSV summaries that can be visualized within VoxKit without rescanning the filesystem.

API

  • AnalyzerManager.list_analyzers: List registered analyzer IDs
  • AnalyzerManager.get_analyzer: Retrieve analyzer instance by ID
  • AnalyzerManager.get_analyzers: Get all registered analyzers
  • DatasetAnalyzer: Abstract base class for all analyzers

Available Analyzers

DefaultAnalyzer (default_analyzer.py) Extracts speaker count and audio file counts per speaker directory. Includes a bar chart visualization for quick dataset overview.

Output Structure

Analyzer output is stored alongside dataset metadata::

~/.voxkit/datasets/{dataset_id}/
├── voxkit_dataset.json           # Dataset metadata
├── {analyzer_name}_summary.csv   # Analyzer output
└── alignments/                   # Alignment outputs

Notes

  • Analyzers run during dataset registration
  • Each analyzer's name property serves as its unique identifier
  • Output is a list of dicts where keys become CSV column headers
  • Custom visualizations can be provided via the visualize method

39class DatasetAnalyzer(ABC):
40    """Abstract base class for dataset analysis methods.
41
42    Subclasses must implement the ``name`` and ``description`` properties and
43    the ``analyze`` method which returns row dictionaries suitable for CSV
44    export.
45    """
46
47    @property
48    @abstractmethod
49    def name(self) -> str:
50        """Display name for this analysis method.
51
52        Returns:
53            str: Human readable name used to register and select the analyzer.
54        """
55
56    @property
57    @abstractmethod
58    def description(self) -> str:
59        """Short description of the analyzer's behavior.
60
61        Returns:
62            str: Brief description suitable for display in UI lists.
63        """
64
65    @abstractmethod
66    def analyze(self, dataset_path: str) -> List[Dict[str, Any]]:
67        """Analyze a dataset and return structured data for CSV export.
68
69        Args:
70            dataset_path (str): Path to the dataset root directory. The
71                analyzer should inspect subdirectories and files below this
72                path.
73
74        Returns:
75            List[Dict[str, Any]]: A list of dictionaries where each
76            dictionary represents a single CSV row. Dictionary keys will be
77            used as CSV column headers.
78        """
79
80    def visualize(self, data: List[Dict[str, Any]]) -> QWidget | None:
81        """Return a QWidget visualizing the analysis data.
82
83        Subclasses may override this method to provide a custom visualization
84        widget. The default implementation returns ``None``, which signals
85        the caller to fall back to a plain table view.
86
87        Args:
88            data: The analysis results as returned by :meth:`analyze`.
89
90        Returns:
91            A QWidget ready for embedding in a dialog, or ``None`` for the
92            default table display.
93        """
94        return None

 25    """Default analyzer extracting speaker and audio file counts per speaker."""
 26
 27    @property
 28    def name(self) -> str:
 29        return "Default"
 30
 31    @property
 32    def description(self) -> str:
 33        return "Speaker count and audio files per speaker"
 34
 35    def analyze(self, dataset_path: str) -> List[Dict[str, Any]]:
 36        """
 37        Return a list of rows with speaker id and audio file count.
 38
 39        Args:
 40            dataset_path (str): Path to the dataset root directory.
 41
 42        Returns:
 43            List[Dict[str, Any]]: Each dict contains ``speaker_id`` and
 44            ``audio_file_count``.
 45        """
 46        results = []
 47        audio_extensions = {".wav", ".flac", ".mp3", ".ogg", ".m4a"}
 48
 49        try:
 50            for entry in os.scandir(dataset_path):
 51                if entry.is_dir():
 52                    speaker_name = entry.name
 53                    audio_files = [
 54                        f
 55                        for f in os.scandir(entry.path)
 56                        if f.is_file() and Path(f.name).suffix.lower() in audio_extensions
 57                    ]
 58
 59                    results.append(
 60                        {"speaker_id": speaker_name, "audio_file_count": len(audio_files)}
 61                    )
 62        except Exception as e:
 63            print(f"Error analyzing dataset: {e}")
 64
 65        return results

 67    def visualize(self, data):
 68        from PyQt6.QtCore import QRectF, Qt
 69        from PyQt6.QtGui import QColor, QFont, QPainter
 70        from PyQt6.QtWidgets import QLabel, QScrollArea, QVBoxLayout, QWidget
 71
 72        entries = []
 73        for row in data:
 74            speaker = str(row.get("speaker_id", ""))
 75            try:
 76                count = int(row.get("audio_file_count", 0))
 77            except (TypeError, ValueError):
 78                count = 0
 79            entries.append((speaker, count))
 80        entries.sort(key=lambda x: x[1], reverse=True)
 81
 82        if not entries:
 83            empty = QLabel("No data to visualize.")
 84            empty.setAlignment(Qt.AlignmentFlag.AlignCenter)
 85            return empty
 86
 87        BAR_HEIGHT = 28
 88        BAR_SPACING = 6
 89        LABEL_WIDTH = 140
 90        PADDING = 16
 91        COUNT_WIDTH = 55
 92        max_count = max(c for _, c in entries) or 1
 93        total_h = PADDING + len(entries) * (BAR_HEIGHT + BAR_SPACING) + PADDING
 94
 95        class _Canvas(QWidget):
 96            def __init__(self, parent=None):
 97                super().__init__(parent)
 98                self.setMinimumHeight(max(total_h, 100))
 99                self.setMinimumWidth(400)
100
101            def paintEvent(self, event):
102                painter = QPainter(self)
103                painter.setRenderHint(QPainter.RenderHint.Antialiasing)
104                bar_area = self.width() - LABEL_WIDTH - PADDING * 2 - COUNT_WIDTH
105                y = PADDING
106
107                for speaker, count in entries:
108                    painter.setPen(QColor("#2c3e50"))
109                    label_font = QFont()
110                    label_font.setPointSize(10)
111                    painter.setFont(label_font)
112                    label_rect = QRectF(PADDING, y, LABEL_WIDTH, BAR_HEIGHT)
113                    fm = painter.fontMetrics()
114                    elided = fm.elidedText(
115                        speaker, Qt.TextElideMode.ElideRight, int(LABEL_WIDTH - 8)
116                    )
117                    painter.drawText(
118                        label_rect,
119                        Qt.AlignmentFlag.AlignRight | Qt.AlignmentFlag.AlignVCenter,
120                        elided,
121                    )
122
123                    bar_x = PADDING + LABEL_WIDTH + 10
124                    bar_w = max((count / max_count) * bar_area, 2)
125                    bar_rect = QRectF(bar_x, y + 3, bar_w, BAR_HEIGHT - 6)
126
127                    color = QColor("#3498db")
128                    h, s, lightness, a = color.getHsl()
129                    assert (
130                        h is not None and s is not None and lightness is not None and a is not None
131                    )
132                    ratio = count / max_count
133                    new_l = int(lightness + (220 - lightness) * (1 - ratio))
134                    color.setHsl(h, s, min(new_l, 240), a)
135
136                    painter.setBrush(color)
137                    painter.setPen(Qt.PenStyle.NoPen)
138                    painter.drawRoundedRect(bar_rect, 4, 4)
139
140                    count_font = QFont()
141                    count_font.setPointSize(9)
142                    count_font.setBold(True)
143                    painter.setFont(count_font)
144                    painter.setPen(QColor("#2c3e50"))
145                    count_rect = QRectF(bar_x + bar_w + 6, y, COUNT_WIDTH, BAR_HEIGHT)
146                    painter.drawText(
147                        count_rect,
148                        Qt.AlignmentFlag.AlignLeft | Qt.AlignmentFlag.AlignVCenter,
149                        str(count),
150                    )
151
152                    y += BAR_HEIGHT + BAR_SPACING
153
154                painter.end()
155
156        container = QWidget()
157        layout = QVBoxLayout(container)
158        layout.setContentsMargins(0, 0, 0, 0)
159        layout.setSpacing(8)
160
161        canvas = _Canvas()
162        scroll = QScrollArea()
163        scroll.setWidget(canvas)
164        scroll.setWidgetResizable(True)
165        scroll.setStyleSheet("QScrollArea { border: none; background: white; }")
166        layout.addWidget(scroll)
167
168        total_speakers = len(entries)
169        total_files = sum(c for _, c in entries)
170        avg = total_files / total_speakers if total_speakers else 0
171        stats = QLabel(
172            f"{total_speakers} speakers  |  {total_files} audio files  |"
173            f"  {avg:.1f} avg files/speaker"
174        )
175        stats.setStyleSheet("color: #7f8c8d; font-size: 12px; font-style: italic; padding: 8px;")
176        stats.setAlignment(Qt.AlignmentFlag.AlignCenter)
177        layout.addWidget(stats)
178
179        return container
